-- [[ КОНСТРУКТОРЫ ]]
local function CreateToggle(name, callback)
    local Btn = Instance.new("TextButton", Container); Btn.Size = UDim2.new(0, 220, 0, 35); Btn.BackgroundColor3 = Color3.fromRGB(40, 40, 55); Btn.Text = name; Btn.TextColor3 = Color3.new(0.8, 0.8, 0.8); Btn.Font = Enum.Font.Gotham; Instance.new("UICorner", Btn)
    local active = false
    Btn.MouseButton1Click:Connect(function() active = not active; Btn.BackgroundColor3 = active and Color3.fromRGB(60, 100, 255) or Color3.fromRGB(40, 40, 55); callback(active) end)
end

local function CreateSlider(name, min, max, start, callback)
    local SliderCont = Instance.new("Frame", Container); SliderCont.Size = UDim2.new(0, 220, 0, 45); SliderCont.BackgroundTransparency = 1
    local Label = Instance.new("TextLabel", SliderCont); Label.Size = UDim2.new(1, 0, 0, 20); Label.Text = name .. ": " .. start; Label.TextColor3 = Color3.new(1,1,1); Label.BackgroundTransparency = 1; Label.Font = Enum.Font.Gotham
    local BG = Instance.new("Frame", SliderCont); BG.Size = UDim2.new(1, 0, 0, 4); BG.Position = UDim2.new(0, 0, 0.7, 0); BG.BackgroundColor3 = Color3.fromRGB(50, 50, 70); Instance.new("UICorner", BG)
    local Fill = Instance.new("Frame", BG); Fill.Size = UDim2.new((start-min)/(max-min), 0, 1, 0); Fill.BackgroundColor3 = Color3.fromRGB(80, 150, 255); Instance.new("UICorner", Fill)
    local function update(input)
        local per = math.clamp((input.Position.X - BG.AbsolutePosition.X) / BG.AbsoluteSize.X, 0, 1)
        Fill.Size = UDim2.new(per, 0, 1, 0); local val = math.floor(min + (max - min) * per); Label.Text = name .. ": " .. val; callback(val)
    end
    BG.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then local con; con = UserInputService.InputChanged:Connect(function(m) update(m) end); local endCon; endCon = UserInputService.InputEnded:Connect(function() con:Disconnect(); endCon:Disconnect() end); update(i) end end)
end

-- [[ ЛОГИКА ОБХОДА И ОСНОВНОЙ ЦИКЛ ]]
local function GetClosest()
    local near, dist = nil, math.huge
    for _, p in pairs(Players:GetPlayers()) do 
        if p ~= Player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then 
            local d = (Player.Character.HumanoidRootPart.Position - p.Character.HumanoidRootPart.Position).Magnitude
            if d < dist then dist = d; near = p.Character end 
        end 
    end
    return near
end

RunService.Heartbeat:Connect(function(dt)
    local char = Player.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChildOfClass("Humanoid")
    local rainbow = Color3.fromHSV(tick() % 5 / 5, 1, 1)

    -- [[ ОБХОД (ANTI-RESET) ]]
    if _G.Config.Noclip then
        for _, p in pairs(char:GetDescendants()) do if p:IsA("BasePart") and p.CanCollide then p.CanCollide = false end end
    end
    if hum then
        local w, d = hum:FindFirstChild("BodyWidthScale"), hum:FindFirstChild("BodyDepthScale")
        if w and _G.Config.ThinArms and w.Value ~= 0.4 then w.Value = 0.4 end
        if d and _G.Config.ThinArms and d.Value ~= 0.4 then d.Value = 0.4 end
    end

    -- [[ ОСНОВНАЯ ЛОГИКА ]]
    if not hrp then return end
    Camera.FieldOfView = _G.Config.FieldOfView
    
    if _G.Config.ESP then 
        local target = GetClosest()
        TargetHighlight.Parent = target or nil
        TargetHighlight.FillColor = rainbow
        TargetHighlight.OutlineColor = rainbow
        TargetHighlight.Enabled = target ~= nil
    else 
        TargetHighlight.Enabled = false 
    end
    
    if _G.Config.TpSpeed and hum and hum.MoveDirection.Magnitude > 0 then 
        hrp.CFrame += hum.MoveDirection * (hum.WalkSpeed * dt * (_G.Config.TpMultiplier - 1)) 
    end
    
    if _G.Config.RGB then 
        for _, p in pairs(char:GetChildren()) do if p:IsA("BasePart") then p.Color = rainbow end end 
    end
end)

UserInputService.JumpRequest:Connect(function()
    if _G.Config.TpJump and Player.Character and Player.Character:FindFirstChild("HumanoidRootPart") then 
        Player.Character.HumanoidRootPart.CFrame += Vector3.new(0, _G.Config.TpJumpPower, 0) 
    end
end)
